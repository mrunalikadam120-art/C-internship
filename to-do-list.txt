#include <iostream>
#include <vector>
#include <string>
using namespace std;

// Structure to represent a task
struct Task {
    string description;
    bool completed;
};

// Function to display the menu
void showMenu() {
    cout << "\nðŸ“‹ To-Do List Manager\n";
    cout << "-------------------------\n";
    cout << "1. Add Task\n";
    cout << "2. View Tasks\n";
    cout << "3. Mark Task as Completed\n";
    cout << "4. Remove Task\n";
    cout << "5. Exit\n";
    cout << "Enter your choice: ";
}

// Function to add a new task
void addTask(vector<Task> &tasks) {
    cin.ignore(); // clear input buffer
    Task newTask;
    cout << "Enter the task description: ";
    getline(cin, newTask.description);
    newTask.completed = false;
    tasks.push_back(newTask);
    cout << "âœ… Task added successfully!\n";
}

// Function to view all tasks
void viewTasks(const vector<Task> &tasks) {
    if (tasks.empty()) {
        cout << "No tasks in your to-do list yet.\n";
        return;
    }

    cout << "\nðŸ“ Your To-Do List:\n";
    cout << "-------------------------\n";
    for (size_t i = 0; i < tasks.size(); ++i) {
        cout << i + 1 << ". " << tasks[i].description
             << " [" << (tasks[i].completed ? "âœ… Completed" : "â³ Pending") << "]\n";
    }
}

// Function to mark a task as completed
void markCompleted(vector<Task> &tasks) {
    if (tasks.empty()) {
        cout << "No tasks to mark as completed.\n";
        return;
    }

    viewTasks(tasks);
    int index;
    cout << "Enter the task number to mark as completed: ";
    cin >> index;

    if (index < 1 || index > tasks.size()) {
        cout << "Invalid task number!\n";
        return;
    }

    if (tasks[index - 1].completed) {
        cout << "Task already completed.\n";
    } else {
        tasks[index - 1].completed = true;
        cout << "ðŸŽ‰ Task marked as completed!\n";
    }
}

// Function to remove a task
void removeTask(vector<Task> &tasks) {
    if (tasks.empty()) {
        cout << "No tasks to remove.\n";
        return;
    }

    viewTasks(tasks);
    int index;
    cout << "Enter the task number to remove: ";
    cin >> index;

    if (index < 1 || index > tasks.size()) {
        cout << "Invalid task number!\n";
        return;
    }

    tasks.erase(tasks.begin() + (index - 1));
    cout << "ðŸ—‘ï¸ Task removed successfully!\n";
}

// Main function
int main() {
    vector<Task> tasks;
    int choice;

    while (true) {
        showMenu();
        cin >> choice;

        switch (choice) {
            case 1:
                addTask(tasks);
                break;
            case 2:
                viewTasks(tasks);
                break;
            case 3:
                markCompleted(tasks);
                break;
            case 4:
                removeTask(tasks);
                break;
            case 5:
                cout << "ðŸ‘‹ Exiting To-Do List Manager. Have a productive day!\n";
                return 0;
            default:
                cout << "Invalid choice. Please try again.\n";
        }
    }
}
